{
	R = 0;

{
	this->R = R;
}

{

	cout << "~Circle()" << endl;

{
	//将传入的 所有属性拷贝到c上。

{
	double cf = 2 * PI * R;
	return cf;
}

	this->R = R;
}

	Circle* c = new Circle(10);
	Circle* b = new Circle();
	Circle a;

	system("pause"); //可用来暂停窗口
	cout << bf << endl;
	cout << sf << endl;
	delete c;
	delete b;
	return 0;

}

构造函数调用规则

  • 如果用户定义有参构造函数，c++不在提供 默认无参构造函数，但是会提供拷贝构造函数

指针数组冒泡排序

#include<iostream>

using namespace std;

void bubbleSort(int *arr,int len);

int main() {
	int arr[] = { 2,4,7,3,7,31,8,4,5,7 };

	int len = sizeof(arr) / sizeof(arr[0]);
	bubbleSort(arr, len);

	for (int i = 0; i < len; i++) {
		cout << *(arr+i) << endl;
	}
	return 0;
}

void bubbleSort(int* arr, int len)
{
	int t;
	for (int i = 0; i < len - 1; i++)
	{
		for (int j = 0; j < len - i - 1; j++) {
			if (arr[j] > arr[j+1]) {
				t = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = t;
			}
		}
	}
}

		m_BedRoom = "卧室";

};
